XML Format:

<cmd>
	<action paranum=0,1,2,...>
		same as action in cmd parameters
	</action>
	<para index=0,1 type=box,file,boxfile,path>
		<box>BOX_NAME</box>
		<file>FILE_NAME</file>
		===============================
		<path>PATH</path>
	</para>
</cmd>

<cmd>
	<action paranum=2>
		set
	</action>
	<para index='0' type='setting_key'>
		...
	</para>
	<para index='1' type='setting_value'>
		...
	</para>
</cmd>
